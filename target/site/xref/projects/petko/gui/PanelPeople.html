425 	public void fillTablePanel_allPeople() {
426 		DefaultTableModel model = (DefaultTableModel) tablePeople.getModel();
427 		model.setRowCount(0);
428 		DateTimeFormatter formatters = DateTimeFormatter.ofPattern("dd-MM-yyyy");
429 		List<Person> ppl = db.getAllPeople();
430 		Object[] row = new Object[8];
431 		for (int i = 0; i < ppl.size(); i++) {
432 			row[0] = ppl.get(i).getId();
433 			row[1] = ppl.get(i).getFirstName();
434 			row[2] = ppl.get(i).getLastName();
435 			row[3] = ppl.get(i).getNickname();
436 			row[4] = ppl.get(i).getCategory();
437 			row[5] = ppl.get(i).getGroup();
438 			row[6] = ppl.get(i).getYearsOld();
439 			if(row[6] instanceof Integer) {
440 				if(Integer.valueOf(row[6].toString()) < 5) {
441 					row[6] = "-";
442 				}
443 			}
444 			row[7] = ppl.get(i).getBirthdayDate().format(formatters);
445 			model.addRow(row);
446 		}
447 		countAllPpl = ppl.size();
448 		labelCountPpl.setText("Showing " + countAllPpl + " out of " + countAllPpl);
449 	}

116 	public void fillTablePanel_NextUpJoined() {
117 		DefaultTableModel model = (DefaultTableModel) tableUpNext.getModel();
118 		model.setRowCount(0);
119 		DateTimeFormatter formatters = DateTimeFormatter.ofPattern("dd-MM-yyyy");
120 		List<Person> nextUp = db.getNextSeven();
121 		Object[] row = new Object[9];
122 		if (nextUp != null) {
123 			for (int i = 0; i < nextUp.size(); i++) {
124 				row[0] = nextUp.get(i).getId();
125 				row[1] = nextUp.get(i).getFirstName();
126 				row[2] = nextUp.get(i).getLastName();
127 				row[3] = nextUp.get(i).getNickname();
128 				row[4] = nextUp.get(i).getCategory();
129 				row[5] = nextUp.get(i).getGroup();
130 				row[6] = nextUp.get(i).getYearsOld();
131 				if(row[6] instanceof Integer) {
132 					if(Integer.valueOf(row[6].toString()) < 5) {
133 						row[6] = "-";
134 					}
135 				}
136 				row[7] = nextUp.get(i).getBirthdayDate().format(formatters);
137 				row[8] = nextUp.get(i).getDaysTill();
138 				model.addRow(row);
139 			}
140 		}

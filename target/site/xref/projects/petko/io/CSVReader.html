13  public class CSVReader {
14  	
15  	private String filename;
16  	private ArrayList<String[]> dbTokens;
17  	
18  	// intiate an instance with the desired file
19  	
20  	public CSVReader(String filename) {
21  		this.filename = filename;
22  		dbTokens = new ArrayList<String[]>();
23  	}
24  	
25  	private void parseLine(String line) {
26  		String[] tokens = line.split(";");
27  		// TODO string split error
28  		
29  		for(int i=0; i < tokens.length; i++) {
30  			if((tokens[i].length() > 0) && (tokens[i].charAt(0) == '"')) {
31  				tokens[i] = tokens[i].substring(1, tokens[i].length() - 1);
32  			}
33  		}
34  		
35  		dbTokens.add(tokens);
36  	}
37  	
38  	public List<String[]> readFile() {
39  
40  		try(BufferedReader bufR = new BufferedReader(new InputStreamReader(
41  				new FileInputStream(this.filename), StandardCharsets.UTF_8))){
42  			String line = bufR.readLine();
43  			while ((line = bufR.readLine()) != null) {
44  				if (line.length() > 0) {
45  					parseLine(line);
46  				}
47  			}
48  			bufR.close();
49  		} catch (FileNotFoundException e) {
50  			e.printStackTrace();
51  		} catch (IOException e) {
52  			e.printStackTrace();
53  		}
54  		return this.dbTokens;
55  	}

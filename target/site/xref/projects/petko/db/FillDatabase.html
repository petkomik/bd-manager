15  	private void clearTable(String name) {
16  		try {
17  			Statement statement = connection.createStatement();
18  			String sql = "DELETE FROM " + name + ";";
19  			statement.execute(sql);
20  		} catch (SQLException e) {
21  			e.printStackTrace();
22  		}
23  	}

25  	public void fillTable() {
26  		try {
27  			connection.setAutoCommit(false);
28  			fillPeople();
29  			connection.commit();
30  			connection.setAutoCommit(true);
31  		
32  		} catch (SQLException e) {
33  			e.printStackTrace();
34  		}
35  	}

37  	public void fillPeople() {
38  		String filename = Parameters.ressourcesDir + Parameters.filenameCSV;
39  		CSVReader csvReader = new CSVReader(filename);
40  		List<String[]> tokens = csvReader.readFile();
41  		clearTable("People");
42  		PreparedStatement prest = null;	
43  		DateTimeFormatter dtf = DateTimeFormatter.ofPattern("dd/MM/yyyy");
44  		DateTimeFormatter dtf2 = DateTimeFormatter.ofPattern("dd-MM-yyyy");
45  
46  
47  		
48  		try {
49  			prest = connection.prepareStatement("INSERT INTO People "
50  					+ "(ID, FName, LName, Nickname, Category, Subgroup, "
51  					+ "Birthday) "
52  					+ "VALUES (?,?,?,?,?,?,?)");
53  		} catch (Exception e) {
54  			System.err.println(e.getClass().getName() + ": " + e.getMessage() );
55  			e.printStackTrace();
56  			System.exit(0);
57  		}
58  		
59  		for(String[] thisLine : tokens) {
60  			try {
61  				prest.setInt(1,Integer.parseInt(thisLine[0]));
62  				prest.setString(2,thisLine[3]);
63  				prest.setString(3,thisLine[4]);
64  				prest.setString(4,thisLine[5]);
65  				prest.setString(5,thisLine[1]);
66  				prest.setString(6,thisLine[2]);
67  				prest.setString(7,LocalDate.parse(thisLine[6], dtf).format(dtf2));
68  				prest.executeUpdate();
69  			} catch (Exception e) {
70  				System.err.println(e.getClass().getName() + ": " + e.getMessage() );
71  				e.printStackTrace();
72  				System.exit(0);
73  			}
74  			
75  		}
76  
77  		
78  	}
